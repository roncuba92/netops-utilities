config vpn ipsec phase1-interface

set interface "port1"
set remote-gw 203.0.113.10
set psksecret "SuperSecreto007!"
set ike-version 2
set proposal des-sha1
set dhgrp 2
set keylife 28800
set dpd on-idle

set nattraversal enable

next
end
config vpn ipsec phase2-interface
edit fgt-pa-ipsec-p2
set phase1name fgt-pa-ipsec
set proposal des-sha1
set pfs enable
set dhgrp 2
set keylifeseconds 3600
set src-subnet 192.168.10.0 255.255.255.0
set dst-subnet 10.20.10.0 255.255.255.0
next
end
config system interface
edit "fgt-pa-ipsec"
set alias "VPN fgt-pa-ipsec"
set type tunnel
set interface "port1"
set ip 169.255.1.1 255.255.255.255
set remote-ip 169.255.1.2 255.255.255.255
set allowaccess ping
next
end
config router static
edit 0
set dst 10.20.10.0 255.255.255.0
set gateway 169.255.1.2
set device "fgt-pa-ipsec"
set distance 10
next
end
config firewall address
edit "fgt-pa-ipsec-fgt-1"
set subnet 192.168.10.0 255.255.255.0
set type ipmask
next
edit "fgt-pa-ipsec-pa-1"
set subnet 10.20.10.0 255.255.255.0
set type ipmask
next
end
config firewall addrgrp
edit "grp-fgt-pa-ipsec-fgt-local"
set member "fgt-pa-ipsec-fgt-1"
next
edit "grp-fgt-pa-ipsec-pa-remote"
set member "fgt-pa-ipsec-pa-1"
next
end
